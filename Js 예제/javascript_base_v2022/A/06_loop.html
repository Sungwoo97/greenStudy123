Loop

for 반복문

      for(초기값; 조건; 증감){
        //반복할 일
      }

while 반복문

      초기값;
      while (조건){
        반복할 일;
        증감;
      }

do while 반복문

      초기값;
      do{
        반복할일;
        증감;       // 무조건 한번은 실행됨
      }while(조건);

forEach 반복문

      배열의 값들 가지고 반복문
      대상은 반드시 배열
      대상, 배열의 각 원소들마다 할 일
      할일 = function(){  }
      대상.forEach(function(각 원소, 인덱스, 전체요소)){ 각 원소들마다 할 일; }
      대상.forEach(function(item, idx, all)){ 각 원소들마다 할 일; }

for in 반복문

      대상은 객체
      객체 = 변수 + 함수;

      for(원소 in 대상){
        반복할일;
      }

for of 반복문

      배열의 값들 가지고 반복문
      대상은 반드시 배열
      for(원소 of 대상){
        //
      }
